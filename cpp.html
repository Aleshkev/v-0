C++ jest jednym z wielu języków programowania. Został stworzony w 1983 roku przez Bjarne'a Stroustrupa.

int fibonacci(long int index, long int start=0)
{
     int f1=start, f2;
     index--;
     f2=fibonacci(start+1);
     for(int i=0, _f2;i<index-1;i++)
     {
         _f2=f2;
         f2=f1+f2;
         f1=_f2;
         cout<<f2<<"\n";
     }
     return f2;
}
